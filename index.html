/* get a list of all files in the "uploads" folder and come up with a new name for the new image */

function newName($dir) {
    
    $files = glob($dir.'*.*');
    $i = 0;

    foreach($files as $file) {
        $file = pathinfo($file);
        $files[$i++] = $file['filename'];
    }

    if(count($files) == 0)
        $name = 1;
    else
        $name = max($files) + 1;
    
    return $name;
}
